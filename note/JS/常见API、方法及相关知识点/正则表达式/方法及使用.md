```js
stringObject.replace(regexp/substr,replacement)

// "<color color_tag='#112233+++#112233'>" ====> "<color color_tag='{{#112233}}+++{{#112233}}'>"

str.replace(/(#.{6})/g,function(){
    return "{{"+arguments[1]+"}}";
})
```
