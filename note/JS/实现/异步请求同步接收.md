如何让异步操作顺序但同步返回结果(比如五个异步请求发出去，按照发送顺序显示收到的结果，如果前一个未执行完就等待)

```js

    let queue = []; // id: {cb, data, haveAccept}
    let uId = 0;
    let acceptId = 0;

    function request(url, methods, data, callback) {
      let id = uId++;
      queue[id] = {};
      queue[id].cb = callback;

      let delayTime = 1000 * Math.random()*2+1;// 模拟用时
      setTimeout(() => { // 假装请求结束，获取异步结果
        queue[id].data = `${id}的结果`;
        queue[id].useTime = delayTime;
        queue[id].haveAccept = true;

        while (acceptId === id && queue[id] && queue[id].haveAccept) {
          queue[id].cb(queue[id].data+'用时：'+queue[id].useTime);
          acceptId++;id++;
        }

      }, delayTime);
    }


    request('','','',(data) => {
      console.log(data);
    })
    request('','','',(data) => {
      console.log(data);
    })
    request('','','',(data) => {
      console.log(data);
    })
    request('','','',(data) => {
      console.log(data);
    })


```
