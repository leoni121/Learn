function throttle(fun, delay) {
  let canRun = true;
  let first = true;
  return function (...args) {
    if(!canRun) return;
    setTimeout(() => {
      fn.apply(this, args);
      canRun = true;
    }, delay)
    if(first) {
      fn.apply(this, args);
      first = false;
    }
  }
}
